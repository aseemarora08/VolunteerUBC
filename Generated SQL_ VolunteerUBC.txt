CREATE TABLE INTERVIEW
(
  InterviewID VARCHAR(50) NOT NULL,
  InterviewDate DATE NOT NULL,
  InterviewTime TIME NOT NULL,
  PRIMARY KEY (InterviewID)
);

CREATE TABLE QUESTION
(
  QuestionID VARCHAR(50) NOT NULL,
  Question VARCHAR(500) NOT NULL,
  QuestionNote VARCHAR(500) NOT NULL,
  PRIMARY KEY (QuestionID)
);

CREATE TABLE TRAINING
(
  TrainingID VARCHAR(50) NOT NULL,
  TrainingDate DATE NOT NULL,
  TrainingStartTime TIME NOT NULL,
  TrainingEndTime TIME NOT NULL,
  PRIMARY KEY (TrainingID)
);

CREATE TABLE INTERVIEW_PROCESS
(
  InterviewID VARCHAR(50) NOT NULL,
  QuestionID VARCHAR(50) NOT NULL,
  PRIMARY KEY (InterviewID, QuestionID),
  FOREIGN KEY (InterviewID) REFERENCES INTERVIEW(InterviewID),
  FOREIGN KEY (QuestionID) REFERENCES QUESTION(QuestionID)
);

CREATE TABLE APPLICANT
(
  ApplicationID VARCHAR(50) NOT NULL,
  Name CHAR(50) NOT NULL,
  Phone# INT NOT NULL,
  Email VARCHAR(50) NOT NULL,
  Availability_for_interview DATE NOT NULL,
  Street VARCHAR(1000) NOT NULL,
  City VARCHAR(100) NOT NULL,
  Province VARCHAR(100) NOT NULL,
  PostalCode VARCHAR(100) NOT NULL,
  InterviewID VARCHAR(50),
  PRIMARY KEY (ApplicationID),
  FOREIGN KEY (InterviewID) REFERENCES INTERVIEW(InterviewID)
);

CREATE TABLE NOTES
(
  NoteID VARCHAR(50) NOT NULL,
  EntryDate DATE NOT NULL,
  Content VARCHAR(500) NOT NULL,
  ApplicationID VARCHAR(50) NOT NULL,
  PRIMARY KEY (NoteID),
  FOREIGN KEY (ApplicationID) REFERENCES APPLICANT(ApplicationID)
);

CREATE TABLE VOLUNTEER
(
  VolunteerID VARCHAR(50) NOT NULL,
  StartDate DATE NOT NULL,
  DailyRate FLOAT NOT NULL,
  ApplicationID VARCHAR(50) NOT NULL,
  PRIMARY KEY (VolunteerID),
  FOREIGN KEY (ApplicationID) REFERENCES APPLICANT(ApplicationID)
);

CREATE TABLE SHIFT
(
  ShiftID VARCHAR(50) NOT NULL,
  ShiftDate DATE NOT NULL,
  Notes VARCHAR(500) NOT NULL,
  ShiftStartTime TIME NOT NULL,
  ShiftEndTime TIME NOT NULL,
  VolunteerID VARCHAR(50) NOT NULL,
  PRIMARY KEY (ShiftID),
  FOREIGN KEY (VolunteerID) REFERENCES VOLUNTEER(VolunteerID)
);

CREATE TABLE TRAINING_SESSION
(
  VolunteerID VARCHAR(50) NOT NULL,
  TrainingID VARCHAR(50) NOT NULL,
  PRIMARY KEY (VolunteerID, TrainingID),
  FOREIGN KEY (VolunteerID) REFERENCES VOLUNTEER(VolunteerID),
  FOREIGN KEY (TrainingID) REFERENCES TRAINING(TrainingID)
);
